  <script>
    async function loadSummary() {
      const res = await fetch("/api/dashboard/summary");
      const json = await res.json();

      if (!json.ok) return;

      document.getElementById("total_cases").textContent = json.data.total_cases;
      document.getElementById("new_cases").textContent = json.data.new_cases;
      document.getElementById("urgent_cases").textContent = json.data.urgent_cases;
      document.getElementById("today_cases").textContent = json.data.today_cases;
    }

    async function loadCases(status = "", priority = "") {
      const params = new URLSearchParams();
      if (status) params.set("status", status);
      if (priority) params.set("priority", priority);
      params.set("limit", "30");

      const res = await fetch("/api/dashboard/cases?" + params.toString());
      const json = await res.json();

      const tbody = document.getElementById("caseTableBody");
      tbody.innerHTML = "";

      if (!json.ok || !json.data || json.data.length === 0) {
        tbody.innerHTML = `<tr><td colspan="7">ไม่พบข้อมูล</td></tr>`;
        return;
      }

      json.data.forEach(item => {
        const tr = document.createElement("tr");
        tr.innerHTML = `
          <td>${item.case_code || "-"}</td>
          <td>${item.full_name || "-"}</td>
          <td>${item.location || "-"}</td>
          <td>${item.phone || "-"}</td>
          <td>${item.status || "-"}</td>
          <td>
            <span class="tag ${item.priority === "urgent" ? "urgent" : "normal"}">
              ${item.priority || "normal"}
            </span>
          </td>
          <td>${item.created_at ? new Date(item.created_at).toLocaleString("th-TH") : "-"}</td>
        `;
        tbody.appendChild(tr);
      });
    }

    loadSummary();
    loadCases();
  </script>
